🔥 Hướng dẫn kết nối Firebase Firestore

1

Tạo project Firebase

Vào console.firebase.google.com → "Add project" → đặt tên → tắt Google Analytics → Tạo project.

2

Tạo Firestore Database

Trong project: Build → Firestore Database → Create database → Chọn "Start in test mode" (thay đổi rules sau) → Chọn vùng: asia-southeast1 (Singapore).

3

Lấy config và thêm vào HTML

Project settings → Your apps → Add app (Web) → Đăng ký app → Copy đoạn config:

// Thêm vào thẻ <head> trong tất cả file HTML
<script type="module">
  import { initializeApp } from "https://www.gstatic.com/firebasejs/10.12.0/firebase-app.js";
  import { getFirestore, collection, getDocs,
    addDoc, updateDoc, deleteDoc, doc }
    from "https://www.gstatic.com/firebasejs/10.12.0/firebase-firestore.js";

  const firebaseConfig = {
    apiKey: "AIzaSy...",
    authDomain: "tuyen-dung-gddt.firebaseapp.com",
    projectId: "tuyen-dung-gddt",
    // ...
  };

  const app = initializeApp(firebaseConfig);
  window.db = getFirestore(app);
  window.fsCol = collection;
  window.fsDocs = getDocs;
  window.fsAdd = addDoc;
  window.fsUpdate = updateDoc;
  window.fsDelete = deleteDoc;
  window.fsDoc = doc;
</script>
4

Cấu trúc Collections

Tạo 3 collection trong Firestore:

📁 pl01_chitieu        // Phụ lục 01
  └─ {docId}: { n, t, c, s, d[], updatedAt }

📁 pl02_chitieu        // Phụ lục 02
  └─ {docId}: { n, t, xa, c, s, d[], updatedAt }

📁 thong_bao           // Thông báo
  └─ {docId}: { tieu_de, loai, noi_dung, ngay, hien_thi }

📁 don_dang_ky         // Đơn ứng viên

5

Đọc & ghi dữ liệu – ví dụ mẫu

// Đọc toàn bộ PL01
async function loadPL01() {
  const snap = await getDocs(collection(db, "pl01_chitieu"));
  return snap.docs.map(d => ({ id: d.id, ...d.data() }));
}

// Thêm mới
async function addRecord(colName, data) {
  await addDoc(collection(db, colName), {
    ...data, updatedAt: new Date().toISOString()
  });
}

// Cập nhật
async function updateRecord(colName, id, data) {
  await updateDoc(doc(db, colName, id), data);
}

// Xoá
async function deleteRecord(colName, id) {
  await deleteDoc(doc(db, colName, id));
}
6

Bảo mật với Firebase Auth

Bật Authentication (Email/Password), tạo tài khoản admin. Cập nhật Firestore Rules:

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Ai cũng đọc được (trang web public)
    match /{document=**} {
      allow read: if true;
    }
    // Chỉ admin đã đăng nhập mới được ghi
    match /{document=**} {
      allow write: if request.auth != null
        && request.auth.token.email == "admin@gddt-dongthap.edu.vn";
    }
  }
}

3

Kết nối từ JavaScript (Supabase SDK)

<script src="https://cdn.jsdelivr.net/npm/@supabase/supabase-js@2"></script>
<script>
  const { createClient } = supabase;
  const SUPABASE_URL = 'https://xxxx.supabase.co';
  const SUPABASE_KEY = 'eyJhbGci...'; // anon public key
  const db = createClient(SUPABASE_URL, SUPABASE_KEY);

  // Đọc dữ liệu PL01
  async function loadPL01() {
    const { data, error } = await db
      .from('pl01_chitieu')
      .select('*')
      .order('stt');
    return data;
  }

  // Thêm mới
  async function addRecord(table, payload) {
    const { data, error } = await db.from(table).insert(payload);
    return data;
  }

  // Cập nhật
  async function updateRecord(table, id, payload) {
    const { data } = await db.from(table).update(payload).eq('id', id);
    return data;
  }

  // Xoá
  async function deleteRecord(table, id) {
    await db.from(table).delete().eq('id', id);
  }
</script>
4

Row Level Security (RLS)

Vào Authentication → Policies → Bật RLS → Thêm policy: mọi người đọc được, chỉ authenticated user mới ghi được.

🐬 Hướng dẫn kết nối MySQL + PHP Backend

1

Tạo database MySQL (cPanel/phpMyAdmin)

CREATE DATABASE tuyen_dung CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;

CREATE TABLE pl01_chitieu (
  id INT AUTO_INCREMENT PRIMARY KEY,
  stt INT, ten_truong VARCHAR(200), cap VARCHAR(50),
  so_chi_tieu INT, d_toan INT, d_ly INT, d_hoa INT,
  d_sinh INT, d_van INT, d_su INT, d_dia INT,
  d_gdktpl INT, d_tanh INT, d_tduc INT, d_ktnn INT,
  d_ktcn INT, d_tin INT, d_gdqp INT, d_am INT,
  d_my INT, d_tieuhoc INT, d_mamnon INT,
  updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
  ON UPDATE CURRENT_TIMESTAMP
);
2

Tạo file PHP API (api.php)

<?php
header('Content-Type: application/json');
header('Access-Control-Allow-Origin: *');
header('Access-Control-Allow-Methods: GET,POST,PUT,DELETE');
header('Access-Control-Allow-Headers: Content-Type,Authorization');

$pdo = new PDO('mysql:host=localhost;dbname=tuyen_dung;charset=utf8mb4',
               'db_user', 'db_password');

$method = $_SERVER['REQUEST_METHOD'];
$table  = $_GET['table'] ?? 'pl01_chitieu';
$id     = $_GET['id'] ?? null;
$body   = json_decode(file_get_contents('php://input'), true);

// Kiểm tra token đơn giản
$token = $_SERVER['HTTP_AUTHORIZATION'] ?? '';
if ($method !== 'GET' && $token !== 'Bearer MY_SECRET_TOKEN') {
  http_response_code(401);
  die(json_encode(['error' => 'Unauthorized']));
}

switch ($method) {
  case 'GET':
    $rows = $pdo->query("SELECT * FROM `$table` ORDER BY stt")->fetchAll(PDO::FETCH_ASSOC);
    echo json_encode($rows);
    break;
  case 'POST':
    $cols = implode(',', array_keys($body));
    $vals = ':' . implode(', :', array_keys($body));
    $stmt = $pdo->prepare("INSERT INTO `$table` ($cols) VALUES ($vals)");
    $stmt->execute($body);
    echo json_encode(['id' => $pdo->lastInsertId()]);
    break;
  case 'PUT':
    $set = implode(', ', array_map(fn($k) => "$k=:$k", array_keys($body)));
    $stmt = $pdo->prepare("UPDATE `$table` SET $set WHERE id=:_id");
    $stmt->execute([...$body, '_id' => $id]);
    echo json_encode(['ok' => true]);
    break;
  case 'DELETE':
    $pdo->prepare("DELETE FROM `$table` WHERE id=?")->execute([$id]);
    echo json_encode(['ok' => true]);
    break;
}
?>
3

Gọi API từ JavaScript

const API = 'https://your-domain.vn/api.php';
const TOKEN = 'Bearer MY_SECRET_TOKEN';

async function loadPL01() {
  const res = await fetch(`${API}?table=pl01_chitieu`);
  return res.json();
}

async function addRecord(table, data) {
  await fetch(`${API}?table=${table}`, {
    method: 'POST',
    headers: { 'Content-Type':'application/json', Authorization: TOKEN },
    body: JSON.stringify(data)
  });
}

async function updateRecord(table, id, data) {
  await fetch(`${API}?table=${table}&id=${id}`, {
    method: 'PUT',
    headers: { 'Content-Type':'application/json', Authorization: TOKEN },
    body: JSON.stringify(data)
  });
}

async function deleteRecord(table, id) {
  await fetch(`${API}?table=${table}&id=${id}`, {
    method: 'DELETE',
    headers: { Authorization: TOKEN }
  });
}
